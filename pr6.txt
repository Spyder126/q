package moreThan60;

import java.io.File;
import java.io.IOException;
import jxl.Cell;
import jxl.CellType;
import jxl.Sheet;
import jxl.Workbook;
import jxl.read.biff.BiffException;



public class Stud_Score {

	public void read() throws BiffException, IOException {
		File inputWorkBook = new File("E:\\Abhijeet tycs 18\\pract5.xls");
		
		int count =0;
		Workbook w =  Workbook.getWorkbook(inputWorkBook);
		Sheet sheet = w.getSheet(0);
		
		for(int j=0;j<sheet.getRows();j++) {
			Cell cell = sheet.getCell(4,j);
			if(cell.getType() == CellType.NUMBER) {
				if(Integer.parseInt(cell.getContents()) > 60) {
					count++;
				}
			}
			
		}
		
		
		System.out.println("Total Number of the Student Scored Above :  " + count);
	}
	
	
	public static void main(String[] args) throws BiffException, IOException {
		// TODO Auto-generated method stub
		Stud_Score ss = new Stud_Score();
		ss.read();

	}

}
